// --- Ball Pool ---
const ballPool = new ObjectPool(
    () => ({
        x: 0, y: 0, startX: 0, startY: 0,
        size: 10, speedY: 0, speedX: 0, baseSpeedY: 0,
        type: 'FASTBALL', timeAlive: 0, active: false,
        hasPaused: false, pauseTimer: 0, invisible: false,
        rotationAngle: 0, rotationSpeed: 0, prevRotationAngle: 0, strikeCounted: false,
        curveDirection: 1, hit: false, spin: 0
    }),
    (b, x, y, type, speedY, speedX, rotationSpeed, curveDir) => {
        b.x = x;
        b.y = y;
        b.prevX = x; // Initialize previous position
        b.prevY = y; // Initialize previous position
        b.startX = x;
        b.startY = y;
        b.size = 10;
        b.speedY = speedY;
        b.speedX = speedX;
        b.baseSpeedY = speedY; // Store initial speedY
        b.type = type;
        b.timeAlive = 0;
        b.active = true;
        b.hasPaused = false;
        b.pauseTimer = 0;
        b.invisible = false;
        b.rotationAngle = 0;
        b.prevRotationAngle = 0;
        b.rotationSpeed = rotationSpeed;
        b.strikeCounted = false;
        b.curveDirection = curveDir;
        b.hit = false;
        b.spin = 0;
    },
    20
);

export function resetGameBalls() {
    // Helper to clear balls on restart
    gameState.balls.forEach(b => ballPool.release(b));
    gameState.balls.length = 0;
}

export function togglePause() {
    if (!gameState.gameActive) return;
    gameState.isPaused = !gameState.isPaused;

    if (gameState.isPaused) {
        pauseOverlay.style.display = 'block';
        settingsBtn.style.display = 'block'; // Show settings when paused
        if (soundManager.audioContext) {
            soundManager.playPause(true); // True = Pausing (Ascending)
            // Do not suspend context so the pause sound can play
        }
        // Fix Jitter: Sync prev positions to current to stop interpolation drift
        gameState.balls.forEach(ball => {
            ball.prevX = ball.x;
            ball.prevY = ball.y;
            ball.prevRotationAngle = ball.rotationAngle;
        });
    } else {
        pauseOverlay.style.display = 'none';
        settingsBtn.style.display = 'none'; // Hide settings when unpaused
        if (soundManager.audioContext) {
            soundManager.playPause(false); // False = Unpausing (Descending)
        }
        gameState.lastTime = performance.now(); // Reset time to prevent huge dt jump
    }
}

export function setPauseEnabled(enabled) {
    if (enabled) {
        pauseBtn.style.opacity = '1';
        pauseBtn.style.pointerEvents = 'auto';
        pauseBtn.style.cursor = 'pointer';
    } else {
        pauseBtn.style.opacity = '0';
        pauseBtn.style.pointerEvents = 'none';
        pauseBtn.style.cursor = 'default';
    }
}

export function startGame() {
    if (!canRestart) return;

    // UI Cleanup (Immediate)
    pauseBtn.style.display = 'flex';
    settingsBtn.style.display = 'none';
    if (unlockablesBtn) unlockablesBtn.style.display = 'none';
    if (leaderboardBtn) leaderboardBtn.style.display = 'none';
    startScreen.style.display = 'none';

    soundManager.stopTheme(); // Stop music immediately on start

    gameState.gameActive = true;
    gameState.isPaused = false; // Ensure game is unpaused

    // Force immediate visual reset (Fix for persistent Game Over tints)
    renderStaticWorld(false);
    renderStaticField();

    pauseOverlay.style.display = 'none'; // Hide pause overlay
    gameState.isGameOver = false; // Reset to Day Mode
    gameState.score = 0;
    gameState.initialHighScore = gameState.highScore; // Capture starting high score
    gameState.strikes = 0;
    gameState.totalPitches = 0;
    gameState.consecutivePerfectHits = 0;
    gameState.consecutiveFouls = 0;
    gameState.consecutiveFouls = 0;
    scoreDisplay.innerText = `Score: ${gameState.score}`;
    gameState.startCoins = gameState.coins; // Track starting coins for session
    coinDisplay.innerText = `${gameState.coins}`; // Initialize coin display
    coinPlusDisplay.style.display = 'none'; // Hide earned coins label
    scoreDisplay.style.visibility = 'visible'; // Show score text
    scoreDisplay.parentElement.style.display = 'flex'; // Ensure container is visible
    coinDisplay.parentElement.style.display = 'none'; // Hide coins during gameplay
    feedbackDisplay.innerText = "";
    pitchIndicator.style.display = 'none';
    startScreen.style.display = 'none';
    resetGameBalls();
    gameState.lastTime = performance.now();
    gameState.accumulator = 0;
    gameState.pitcher.state = PITCHER_STATES.IDLE;
    gameState.pitcher.state = PITCHER_STATES.IDLE;
    gameState.pitcher.timer = 0;
    gameState.showFireworks = false; // Reset fireworks flag
    gameState.timeScale = 1.0; // Reset time scale
    gameState.bossModeActive = false; // Reset Boss Mode
    gameState.pitcher.skinColor = '#ffccaa'; // Reset Skin Color

    // Reset Cursor (Space Theme uses pointer on start screen)
    const gameCanvas = document.querySelector('canvas');
    if (gameCanvas) gameCanvas.style.cursor = 'default';
    gameState.pitcher.scale = 1.0; // Reset Scale

    // Position Reset (Fix for restarting during Boss Transition)
    gameState.pitcher.x = GAME_WIDTH / 2 - gameState.pitcher.width / 2;
    gameState.pitcher.y = 150 + playOffsetY;
    gameState.perfectHitTimer = 0; // Reset Animation Timer

    // Reset Tutorial if it's a fresh start (score 0)
    // Reset Tutorial if it's a fresh start (score 0) AND hasn't been seen yet
    if (gameState.score === 0 && !gameState.tutorialSeen) {
        gameState.showTutorial = true;
    } else {
        gameState.showTutorial = false;
    }

    setPauseEnabled(true);

    // requestAnimationFrame(gameLoop); // Handled by main.js
}

export function gameOver() {
    gameState.gameActive = false;
    gameState.isGameOver = true; // Switch to Night Mode
    canRestart = false; // Prevent immediate restart

    // Check for High Score
    // Use initialHighScore to ensure we only celebrate if they BEAT the previous session's record
    // Ties do not trigger celebration.
    if (gameState.score > gameState.initialHighScore && gameState.score > 10) {
        gameState.highScore = gameState.score;
        localStorage.setItem('baseball_high_score', gameState.highScore);

        gameState.showFireworks = true;
        fireworksManager.reset();
    }

    settingsBtn.style.display = 'block'; // Show settings on Game Over
    if (unlockablesBtn) unlockablesBtn.style.display = 'flex'; // Show unlockables on Game Over
    if (leaderboardBtn) leaderboardBtn.style.display = 'block'; // Show leaderboard on Game Over

    renderStaticWorld(true); // Render Night Mode

    // Update Start Screen with Game Over message and Score
    startScreen.innerHTML = `
        <span style="color: #ff0000; font-size: 1.2em;">GAME OVER</span><br>
        <span style="font-size: 1.0em; display: block; margin-top: 10px; animation: none; color: #fff;">SCORE: ${gameState.score}</span>
        <span id="restart-msg" style="font-size: 0.8em; display: block; margin-top: 20px; animation: ${gameState.settings?.noFlash ? 'none' : 'blink 1s infinite'}; color: #fff;">TAP TO RETRY</span>
    `;
    startScreen.style.display = 'block';

    scoreDisplay.parentElement.style.display = 'flex'; // Ensure container is visible
    scoreDisplay.style.visibility = 'hidden'; // Hide score text on Game Over
    coinDisplay.parentElement.style.display = 'flex'; // Show coins on Game Over
    feedbackDisplay.innerText = "";
    soundManager.playTheme(); // Play music

    // Submit High Score to Leaderboard (Auto-submit if name exists)
    leaderboardUI.checkAndSubmitScore(gameState.score);

    // Enable restart after 1 second
    setTimeout(() => {
        canRestart = true;
    }, 1000);
    pauseBtn.style.display = 'none'; // Hide pause button
    setPauseEnabled(true);

    // Coin Animation
    const earnedCoins = gameState.coins - (gameState.startCoins || 0);
    if (earnedCoins > 0) {
        // Show +X label
        coinPlusDisplay.innerText = `+${earnedCoins}`;
        coinPlusDisplay.style.display = 'block';

        // Animate Counter
        let currentDisplay = gameState.startCoins || 0;
        const targetDisplay = gameState.coins;
        const duration = 1500; // 1.5 seconds
        const startTime = performance.now();

        function animateCoins(time) {
            const elapsed = time - startTime;
            const progress = Math.min(elapsed / duration, 1.0);

            // Ease Out Cubic
            const ease = 1 - Math.pow(1 - progress, 3);

            currentDisplay = Math.floor((gameState.startCoins || 0) + (earnedCoins * ease));
            coinDisplay.innerText = `${currentDisplay}`;

            if (progress < 1.0) {
                requestAnimationFrame(animateCoins);
            } else {
                coinDisplay.innerText = `${targetDisplay}`; // Ensure final value is exact
                // Hide +X label after a short delay
                setTimeout(() => {
                    coinPlusDisplay.style.display = 'none';
                }, 1000);
            }
        }
        requestAnimationFrame(animateCoins);
    }
}

export function swing() {
    if (gameState.isPaused) return; // Prevent swing while paused

    if (!gameState.gameActive) {
        if (!canRestart) return;

        // Initialize sound context and wait for it to be ready
        soundManager.init().then(() => {
            startGame();
        });
        return;
    }

    const now = Date.now();
    if (now - gameState.lastSwingTime < SWING_COOLDOWN) return;

    if (!gameState.bat.swinging) {
        gameState.lastSwingTime = now;
        gameState.bat.swinging = true;
        gameState.bat.angle = Math.PI; // Start from Left
        soundManager.playSwing();
    }
}

export function update(dt) {
    // Pre-calculate dt normalization (optimization: avoid repeated division)
    const dtNorm = dt / 16.667;

    if (gameState.balls.length === 0) {
        gameState.pitcher.timer += dt;
    }

    // Update Perfect Hit Animation Timer
    if (gameState.perfectHitTimer > 0) {
        gameState.perfectHitTimer -= dt;
    }

    if (gameState.pitcher.state === PITCHER_STATES.IDLE) {
        // First pitch delay: 1.5s
        // Fouls/Strikes: 1.0s delay
        // Hits/Others: 0s (Instant)
        let delay = 0;
        if (gameState.totalPitches === 0) delay = 1500;
        else if (gameState.lastPitchResult === 'STRIKE' || gameState.lastPitchResult === 'FOUL') delay = 1000;

        if (gameState.pitcher.timer > delay) {
            gameState.pitcher.state = PITCHER_STATES.WINDUP;
            setPauseEnabled(false); // Disable pause when pitch begins
            gameState.pitcher.timer = 0;

            const rand = Math.random();

            // Increment pitch count
            gameState.totalPitches++;

            // Disable tutorial after first pitch is thrown
            if (gameState.totalPitches > 1) {
                gameState.showTutorial = false;
                gameState.tutorialSeen = true; // Mark tutorial as seen
            }

            if ((gameState.totalPitches === 1 || gameState.score === 0) && !gameState.tutorialSeen) {
                // TUTORIAL PITCH: Force Curveball with Long Windup
                gameState.pitcher.nextPitch = PITCH_TYPES.CURVEBALL;
                gameState.pitcher.windupDuration = 4000; // 4 seconds real time
            } else {
                // Standard Progression
                gameState.pitcher.nextPitch = getNextPitchType(gameState.score);
            }

            const pitchConfig = PITCH_DEFINITIONS[gameState.pitcher.nextPitch];

            // Only override windup if NOT in tutorial (tutorial sets its own duration)
            if (!gameState.showTutorial) {
                // If it's the test pitch, it will use the config value (2500) we just set
                gameState.pitcher.windupDuration = pitchConfig.windupDuration || 1000;
            }

            if (pitchConfig.chargeSound && soundManager[pitchConfig.chargeSound]) {
                soundManager[pitchConfig.chargeSound]();
            }

            // Color Blind Mode check moved to update loop for reactivity
        }
    }

    // --- Boss Battle Transition (Score 71 - Start of Hybrids) ---
    if (gameState.score >= 71 && !gameState.bossModeActive && gameState.pitcher.state === PITCHER_STATES.IDLE) {
        gameState.bossModeActive = true;
        gameState.pitcher.state = PITCHER_STATES.WALK_OFF;
        gameState.pitcher.timer = 0;

        // Reward: Clear Strikes
        gameState.strikes = 0;
        gameState.consecutiveFouls = 0;

        // Visual & Audio Feedback
        // Using DIVs to force block-level centering
        feedbackDisplay.innerHTML = "<div style='width:100%; text-align:center'>STRIKES</div><div style='width:100%; text-align:center'>CLEARED!</div>";
        feedbackDisplay.className = "strikes-cleared"; // Unique Neon Green Logic
        soundManager.playStrikesCleared();

        // Clear message after 4 seconds (Spectacular Moment)
        setTimeout(() => {
            if (feedbackDisplay.innerText.includes("CLEARED")) {
                feedbackDisplay.innerText = "";
                feedbackDisplay.className = "";
            }
        }, 4000);
    }

    // --- Pitcher Logic ---
    if (gameState.pitcher.state === PITCHER_STATES.WALK_OFF) {
        gameState.pitcher.timer += dt;
        // Walk off to the left
        gameState.pitcher.x -= 0.1 * dt; // Speed of walk

        // Bobbing animation for walk
        if (Math.floor(gameState.pitcher.timer / 200) % 2 === 0) {
            gameState.pitcher.y = 150 + playOffsetY - 2;
        } else {
            gameState.pitcher.y = 150 + playOffsetY + 2;
        }

        // Check if off screen
        if (gameState.pitcher.x < -50) {
            // Transition to Boss Walk On
            gameState.pitcher.state = PITCHER_STATES.WALK_ON;
            gameState.pitcher.x = GAME_WIDTH + 50; // Start from right side

            // Apply Boss Visuals
            // gameState.pitcher.baseColor = '#4B0082'; // Indigo/Dark Purple for Boss (REMOVED)
            gameState.pitcher.skinColor = '#5C4033'; // Darker Skin
            gameState.pitcher.scale = 1.0; // Same size as original
        }
    } else if (gameState.pitcher.state === PITCHER_STATES.WALK_ON) {
        gameState.pitcher.timer += dt;
        // Walk on from the right
        gameState.pitcher.x -= 0.1 * dt;

        // Bobbing animation
        if (Math.floor(gameState.pitcher.timer / 200) % 2 === 0) {
            gameState.pitcher.y = 150 + playOffsetY - 2;
        } else {
            gameState.pitcher.y = 150 + playOffsetY + 2;
        }

        // Check if back at center
        if (gameState.pitcher.x <= GAME_WIDTH / 2 - gameState.pitcher.width / 2) {
            gameState.pitcher.x = GAME_WIDTH / 2 - gameState.pitcher.width / 2;
            gameState.pitcher.state = PITCHER_STATES.IDLE;
            gameState.pitcher.timer = 0;
            // Boss is ready!
        }
    } else if (gameState.pitcher.state === PITCHER_STATES.WINDUP) {
        gameState.pitcher.timer += dt;

        // Continuous Color Blind Mode Check
        const pitchConfig = PITCH_DEFINITIONS[gameState.pitcher.nextPitch];
        if (gameState.settings?.colorBlindMode) {
            if (pitchIndicator.style.display !== 'block') {
                pitchIndicator.style.display = 'block';
                pitchIndicator.style.textShadow = '2px 2px 0 #000';
                pitchIndicatorVisible = true; // Track visibility
            }

            let newHtml = '';
            if (pitchConfig.colors.length > 1 && pitchConfig.colorName.includes('/')) {
                // Hybrid Pitch: Split text and colorize parts
                const parts = pitchConfig.colorName.split('/');
                parts.forEach((part, index) => {
                    const color = pitchConfig.colors[index] || '#ffffff';
                    newHtml += `<span style="color: ${color}">${part}</span>`;
                    if (index < parts.length - 1) {
                        newHtml += '<span style="color: #ffffff">/</span>';
                    }
                });
            } else {
                // Single Color Pitch
                newHtml = `<span style="color: ${pitchConfig.colors[0]}">${pitchConfig.colorName}</span>`;
            }

            // Only update DOM if content changed
            if (pitchIndicator.innerHTML !== newHtml) {
                pitchIndicator.innerHTML = newHtml;
            }
        } else {
            if (pitchIndicator.style.display !== 'none') {
                pitchIndicator.style.display = 'none';
                pitchIndicatorVisible = false; // Track visibility
            }
        }

        if (gameState.pitcher.nextPitch === PITCH_TYPES.SUPER_ZIGZAG && gameState.pitcher.timer > 1250 && !gameState.pitcher.chargeSoundPlayed) {
            soundManager.playZigZagCharge();
            gameState.pitcher.chargeSoundPlayed = true;
        }

        // SUPER_CURVE Mid-Windup Sound Trigger
        if (gameState.pitcher.nextPitch === PITCH_TYPES.SUPER_CURVE && gameState.pitcher.timer > 1250 && !gameState.pitcher.chargeSoundPlayed) {
            soundManager.playCurveCharge(); // Plays the "Wobbly" sound second
            gameState.pitcher.chargeSoundPlayed = true;
        }

        // STOP_CURVE Mid-Windup Sound Trigger
        if (gameState.pitcher.nextPitch === PITCH_TYPES.STOP_CURVE && gameState.pitcher.timer > 1250 && !gameState.pitcher.chargeSoundPlayed) {
            soundManager.playCurveCharge();
            gameState.pitcher.chargeSoundPlayed = true;
        }



        // STOP_ZIGZAG Mid-Windup Sound Trigger
        // Green Sound = Alien Warble
        if (gameState.pitcher.nextPitch === PITCH_TYPES.STOP_ZIGZAG && gameState.pitcher.timer > 1250 && !gameState.pitcher.chargeSoundPlayed) {
            soundManager.playZigZagCharge();
            gameState.pitcher.chargeSoundPlayed = true;
        }

        // GHOST_ZIGZAG Mid-Windup Sound Trigger
        // Green Sound = Alien Warble
        if (gameState.pitcher.nextPitch === PITCH_TYPES.GHOST_ZIGZAG && gameState.pitcher.timer > 1250 && !gameState.pitcher.chargeSoundPlayed) {
            soundManager.playZigZagCharge();
            gameState.pitcher.chargeSoundPlayed = true;
        }

        // GHOST_CHANGEUP Mid-Windup Sound Trigger
        // Orange Sound = Changeup (Square Wave)
        if (gameState.pitcher.nextPitch === PITCH_TYPES.GHOST_CHANGEUP && gameState.pitcher.timer > 1250 && !gameState.pitcher.chargeSoundPlayed) {
            soundManager.playChangeCharge();
            gameState.pitcher.chargeSoundPlayed = true;
        }

        // CURVE_SPLITTER Mid-Windup Sound Trigger
        // Blue Sound = Splitter
        if (gameState.pitcher.nextPitch === PITCH_TYPES.CURVE_SPLITTER && gameState.pitcher.timer > 1250 && !gameState.pitcher.chargeSoundPlayed) {
            soundManager.playSplitterCharge();
            gameState.pitcher.chargeSoundPlayed = true;
        }

        // ZIGZAG_SPLITTER Mid-Windup Sound Trigger
        // Blue Sound = Splitter
        if (gameState.pitcher.nextPitch === PITCH_TYPES.ZIGZAG_SPLITTER && gameState.pitcher.timer > 1250 && !gameState.pitcher.chargeSoundPlayed) {
            soundManager.playSplitterCharge();
            gameState.pitcher.chargeSoundPlayed = true;
        }

        // CURVE_CHANGEUP Mid-Windup Sound Trigger
        // Gold Sound = Changeup
        if (gameState.pitcher.nextPitch === PITCH_TYPES.CURVE_CHANGEUP && gameState.pitcher.timer > 1250 && !gameState.pitcher.chargeSoundPlayed) {
            soundManager.playChangeCharge();
            gameState.pitcher.chargeSoundPlayed = true;
        }

        // SPLITTER_CHANGEUP Mid-Windup Sound Trigger
        // Gold Sound = Changeup
        if (gameState.pitcher.nextPitch === PITCH_TYPES.SPLITTER_CHANGEUP && gameState.pitcher.timer > 1250 && !gameState.pitcher.chargeSoundPlayed) {
            soundManager.playChangeCharge();
            gameState.pitcher.chargeSoundPlayed = true;
        }

        // ZIGZAG_CHANGEUP Mid-Windup Sound Trigger
        // Gold Sound = Changeup
        if (gameState.pitcher.nextPitch === PITCH_TYPES.ZIGZAG_CHANGEUP && gameState.pitcher.timer > 1250 && !gameState.pitcher.chargeSoundPlayed) {
            soundManager.playChangeCharge();
            gameState.pitcher.chargeSoundPlayed = true;
        }

        // GHOST_STOP Mid-Windup Sound Trigger
        // Purple Sound = Stop (Evil Laugh)
        if (gameState.pitcher.nextPitch === PITCH_TYPES.GHOST_STOP && gameState.pitcher.timer > 1250 && !gameState.pitcher.chargeSoundPlayed) {
            soundManager.playEvilLaugh();
            gameState.pitcher.chargeSoundPlayed = true;
        }

        // GHOST_SUPER Mid-Windup Sound Trigger
        // Red Sound = Super
        if (gameState.pitcher.nextPitch === PITCH_TYPES.GHOST_SUPER && gameState.pitcher.timer > 1250 && !gameState.pitcher.chargeSoundPlayed) {
            soundManager.playSuperCharge();
            gameState.pitcher.chargeSoundPlayed = true;
        }

        // STOP_SUPER Mid-Windup Sound Trigger
        // Red Sound = Super
        if (gameState.pitcher.nextPitch === PITCH_TYPES.STOP_SUPER && gameState.pitcher.timer > 1250 && !gameState.pitcher.chargeSoundPlayed) {
            soundManager.playSuperCharge();
            gameState.pitcher.chargeSoundPlayed = true;
        }

        // SUPER_SPLITTER Mid-Windup Sound Trigger
        // Blue Sound = Splitter
        if (gameState.pitcher.nextPitch === PITCH_TYPES.SUPER_SPLITTER && gameState.pitcher.timer > 1250 && !gameState.pitcher.chargeSoundPlayed) {
            soundManager.playSplitterCharge();
            gameState.pitcher.chargeSoundPlayed = true;
        }

        // STOP_CHANGEUP Mid-Windup Sound Trigger
        // Gold Sound = Changeup
        if (gameState.pitcher.nextPitch === PITCH_TYPES.STOP_CHANGEUP && gameState.pitcher.timer > 1250 && !gameState.pitcher.chargeSoundPlayed) {
            soundManager.playChangeCharge();
            gameState.pitcher.chargeSoundPlayed = true;
        }

        // GHOST_SPLITTER Mid-Windup Sound Trigger
        if (gameState.pitcher.nextPitch === PITCH_TYPES.GHOST_SPLITTER && gameState.pitcher.timer > 1250 && !gameState.pitcher.chargeSoundPlayed) {
            soundManager.playSplitterCharge();
            gameState.pitcher.chargeSoundPlayed = true;
        }

        // STOP_SPLITTER Mid-Windup Sound Trigger
        if (gameState.pitcher.nextPitch === PITCH_TYPES.STOP_SPLITTER && gameState.pitcher.timer > 1250 && !gameState.pitcher.chargeSoundPlayed) {
            soundManager.playSplitterCharge();
            gameState.pitcher.chargeSoundPlayed = true;
        }

        // GHOST_CURVE Mid-Windup Sound Trigger
        if (gameState.pitcher.nextPitch === PITCH_TYPES.GHOST_CURVE && gameState.pitcher.timer > 1250 && !gameState.pitcher.chargeSoundPlayed) {
            soundManager.playCurveCharge();
            gameState.pitcher.chargeSoundPlayed = true;
        }

        if (gameState.pitcher.timer > gameState.pitcher.windupDuration) {
            gameState.pitcher.state = PITCHER_STATES.IDLE;
            gameState.pitcher.timer = 0;
            gameState.pitcher.chargeSoundPlayed = false;

            // Hide Pitch Indicator immediately on release
            pitchIndicator.style.display = 'none';

            // Reset Tutorial State when ball is thrown
            if (gameState.showTutorial) {
                gameState.showTutorial = false;
            }

            // Speed Scaling (Infinite)
            // Phase 1 (0-60): +2% per point (0.6 -> 1.8)
            // Phase 2 (61-120): +1% per point (1.8 -> 2.4)
            // Phase 3 (121+): +0.5% per point (2.4 -> Infinity)
            let speedMultiplier = 0.6;

            if (gameState.score <= 60) {
                speedMultiplier += gameState.score * 0.02;
            } else if (gameState.score <= 120) {
                speedMultiplier += 1.2; // Max of Phase 1 (60 * 0.02)
                speedMultiplier += (gameState.score - 60) * 0.01;
            } else {
                speedMultiplier += 1.2; // Max of Phase 1
                speedMultiplier += 0.6; // Max of Phase 2 (60 * 0.01)
                speedMultiplier += (gameState.score - 120) * 0.001; // Uncapped (Reduced to +0.1%)
            }
            const pitchConfig = PITCH_DEFINITIONS[gameState.pitcher.nextPitch];
            let speedY = pitchConfig.baseSpeed || 0.40;
            let speedX = 0;
            let rotationSpeed = pitchConfig.rotationSpeed || 0.2;

            speedY *= speedMultiplier;

            // Randomize curve direction
            const curveDir = Math.random() > 0.5 ? 1 : -1;

            const ball = ballPool.get(
                gameState.pitcher.x + gameState.pitcher.width / 2 - 5,
                gameState.pitcher.y + gameState.pitcher.height,
                gameState.pitcher.nextPitch,
                speedY * dt,
                speedX,
                rotationSpeed,
                curveDir
            );
            gameState.balls.push(ball);

            if (pitchConfig.throwSound && soundManager[pitchConfig.throwSound]) {
                soundManager[pitchConfig.throwSound]();
            } else {
                soundManager.playPitch();
            }
        }
    }

    for (let i = gameState.balls.length - 1; i >= 0; i--) {
        const ball = gameState.balls[i];

        if (ball.hit) {
            ball.prevX = ball.x;
            ball.prevY = ball.y;
            ball.prevRotationAngle = ball.rotationAngle;
            ball.x += ball.speedX;
            ball.y += ball.speedY;
            ball.invisible = false;
            if (ball.spin) {
                ball.speedX += ball.spin * dtNorm;
            }
            ball.rotationAngle += ball.rotationSpeed * 3 * dtNorm;

            // Hide indicator on hit (optimized: only update DOM if visible)
            if (pitchIndicatorVisible) {
                pitchIndicator.style.display = 'none';
                pitchIndicatorVisible = false;
            }
        } else {
            ball.timeAlive += dt;

            // Delegate specific pitch behavior
            if (PitchBehaviors[ball.type]) {
                const shouldSkipMovement = PitchBehaviors[ball.type](ball, dt);
                if (shouldSkipMovement) continue;
            }

            ball.prevRotationAngle = ball.rotationAngle;
            ball.rotationAngle += ball.rotationSpeed * dtNorm;

            ball.prevX = ball.x;
            ball.prevY = ball.y;
            ball.y += ball.speedY;

            // Check for Missed Ball (Strike)
            if (!ball.hit && !ball.strikeCounted && ball.y > gameState.bat.pivotY + 60) {
                ball.strikeCounted = true;
                gameState.lastPitchResult = 'STRIKE';
                gameState.strikes++;
                gameState.consecutivePerfectHits = 0;
                feedbackDisplay.innerText = "STRIKE!";
                feedbackDisplay.className = "foul-text";
                soundManager.playStrike();
                setPauseEnabled(true);

                // Hide indicator on strike (optimized: only update DOM if visible)
                if (pitchIndicatorVisible) {
                    pitchIndicator.style.display = 'none';
                    pitchIndicatorVisible = false;
                }

                gameState.consecutiveFouls = 0;

                if (gameState.strikes >= 3) {
                    gameOver();
                } else {
                    setTimeout(() => {
                        if (feedbackDisplay.innerText.includes("STRIKE")) {
                            feedbackDisplay.innerText = "";
                            feedbackDisplay.className = "";
                        }
                    }, 1000);
                }
            }
        }

        if (ball.timeAlive > 10000) {
            ballPool.release(ball);
            gameState.balls.splice(i, 1);
            continue;
        }

        if (ball.y > GAME_HEIGHT || ball.y < -50 || ball.x < -50 || ball.x > GAME_WIDTH + 50) {
            ballPool.release(ball);
            gameState.balls.splice(i, 1);
            if (gameState.balls.length === 0) setPauseEnabled(true);
        }
    }

    if (gameState.bat.swinging) {
        gameState.bat.angle += gameState.bat.swingSpeed * (dt / 16);
        if (gameState.bat.angle <= -Math.PI) {
            gameState.bat.angle = Math.PI;
            gameState.bat.swinging = false;
        }
    }

    // Collision Detection
    if (gameState.bat.swinging) {
        for (let i = gameState.balls.length - 1; i >= 0; i--) {
            const ball = gameState.balls[i];
            if (!ball.active) continue;

            const dx = ball.x - gameState.bat.pivotX;
            const dy = ball.y - gameState.bat.pivotY;
            const dist = Math.sqrt(dx * dx + dy * dy);

            const ballAngle = Math.atan2(dy, dx);

            let batAngleNorm = gameState.bat.angle % (Math.PI * 2);
            if (batAngleNorm > Math.PI) batAngleNorm -= Math.PI * 2;
            let angleDiff = Math.abs(batAngleNorm - ballAngle);
            if (angleDiff > Math.PI) angleDiff = Math.PI * 2 - angleDiff;

            const isAngleMatch = angleDiff < 1.2;
            const isBarrelHit = dist > 0 && dist < 70;
            const isBatFacingPlate = true;
            const isBallInFront = ball.x > gameState.bat.pivotX - 10;
            const isBallHighEnough = ball.y < gameState.bat.pivotY + 30;

            if (isAngleMatch && isBarrelHit && isBatFacingPlate && isBallInFront && isBallHighEnough) {
                // console.log(`Hit! Dist: ${dist.toFixed(2)} (Sweet Spot: 25-55)`);
                ball.active = false;
                setPauseEnabled(true);

                if (ball.strikeCounted) {
                    gameState.strikes--;
                    ball.strikeCounted = false;
                    if (!gameState.gameActive) {
                        gameState.gameActive = true;
                        startScreen.style.display = 'none';
                        scoreDisplay.style.display = 'block';
                        soundManager.stopTheme();
                    }
                }

                const hitAngle = gameState.bat.angle;

                const isInnerBarrel = dist >= 25 && dist < 45;
                const isOuterBarrel = dist >= 45 && dist < 58;
                const isTip = dist >= 58 && dist < 70;

                const isTimingGood = hitAngle < 0.5 && hitAngle > -3.0;
                const isGoodInput = (isInnerBarrel && isTimingGood) || isOuterBarrel || isTip;

                // logDebug comment removed (dead code cleanup)

                let zoneLabel = "Handle (Foul)";
                if (isInnerBarrel) zoneLabel = "Inner (Timing)";
                if (isOuterBarrel) zoneLabel = "Outer (Center)";
                if (isTip) zoneLabel = "Tip (Left)";

                // logDebug(`Hit! Dist: ${dist.toFixed(2)} [${zoneLabel}]\nAngle: ${angleLabel} (${hitAngle.toFixed(2)})`);

                const HIT_SPEED = 12 * (dt / 16.667);
                const SAFE_MARGIN = 0.08;

                if (isGoodInput) {
                    let outputAngle;
                    const jitter = (Math.random() - 0.5) * 0.4;

                    if (isTip) {
                        const spread = (Math.random() - 0.5) * 0.4;
                        outputAngle = FAIR_ANGLE_LEFT + 0.15 + spread;
                        feedbackDisplay.innerText = "GOOD HIT!";
                        feedbackDisplay.className = "";

                    } else if (isOuterBarrel) {
                        const spread = (Math.random() - 0.5) * 0.3;
                        outputAngle = CENTER_ANGLE + spread;
                        feedbackDisplay.innerText = "PERFECT HIT!";
                        feedbackDisplay.className = "perfect-hit";

                    } else {
                        const clampedInput = Math.max(-2.0, Math.min(0.5, hitAngle));
                        let inputRatio = (clampedInput - (-1.0)) / (0.5 - (-1.0));
                        inputRatio = Math.max(0, Math.min(1, inputRatio));

                        outputAngle = CENTER_ANGLE + (inputRatio * (FAIR_ANGLE_RIGHT - CENTER_ANGLE));
                        outputAngle += jitter;
                        feedbackDisplay.innerText = "GOOD HIT!";
                        feedbackDisplay.className = "";
                    }

                    outputAngle = Math.max(FAIR_ANGLE_LEFT + SAFE_MARGIN, Math.min(FAIR_ANGLE_RIGHT - SAFE_MARGIN, outputAngle));

                    ball.speedX = Math.cos(outputAngle) * HIT_SPEED;
                    ball.speedY = Math.sin(outputAngle) * HIT_SPEED;

                    if (isOuterBarrel) {
                        gameState.lastPitchResult = 'HIT';
                        gameState.consecutivePerfectHits++;
                        let points = 1;
                        if (gameState.consecutivePerfectHits === 2) points = 2;
                        if (gameState.consecutivePerfectHits >= 3) points = 3;

                        gameState.score += points;

                        // Award Coins for Perfect Hit
                        gameState.coins += points; // 2 or 3 coins
                        localStorage.setItem('baseball_coins', gameState.coins);
                        coinDisplay.innerText = `${gameState.coins}`;

                        if (points > 1) {
                            scorePlusDisplay.innerText = `+${points}`;
                            scorePlusDisplay.style.display = "block";

                            // Apply flashing if not disabled
                            // Force noFlash to false if it was accidentally enabled by default
                            if (gameState.settings && gameState.settings.noFlash === undefined) gameState.settings.noFlash = false;

                            if (!gameState.settings?.noFlash) {
                                scorePlusDisplay.classList.add('perfect-hit');
                            } else {
                                scorePlusDisplay.classList.remove('perfect-hit');
                            }

                            setTimeout(() => {
                                scorePlusDisplay.style.display = "none";
                                scorePlusDisplay.classList.remove('perfect-hit');
                            }, 1000);
                        }

                        feedbackDisplay.innerText = "PERFECT HIT!";
                        if (!gameState.settings?.noFlash) {
                            feedbackDisplay.className = "perfect-hit";
                        } else {
                            feedbackDisplay.className = "";
                        }
                        soundManager.playPerfectHit();

                        // Trigger Console Animation - Sync with text (1000ms)
                        // User Request: Only trigger on +3 consecutive perfect hits and beyond
                        if (gameState.consecutivePerfectHits >= 3) {
                            gameState.perfectHitTimer = 1000;
                        }
                    } else {
                        gameState.lastPitchResult = 'HIT';
                        gameState.consecutivePerfectHits = 0;
                        gameState.score++;

                        // Award Coin for Good Hit
                        gameState.coins++;
                        localStorage.setItem('baseball_coins', gameState.coins);
                        coinDisplay.innerText = `${gameState.coins}`;
                        feedbackDisplay.innerText = "GOOD HIT!";
                        feedbackDisplay.className = "";
                        soundManager.playHit();
                    }

                    gameState.consecutiveFouls = 0;
                    scoreDisplay.innerText = `Score: ${gameState.score}`;

                    if (gameState.score > gameState.highScore) {
                        gameState.highScore = gameState.score;
                        localStorage.setItem('baseball_high_score', gameState.highScore);

                        // Optimization: Skip expensive static render for Space Theme during gameplay
                        // Space Theme uses dynamic UI/Satellite for high score, so re-rendering the planet/cockpit is waste.
                        if (themeManager.currentTheme.id !== 'SPACE') {
                            renderStaticWorld(false);
                        }
                    }

                    setTimeout(() => {
                        if (feedbackDisplay.innerText.includes("HIT")) {
                            feedbackDisplay.innerText = "";
                            feedbackDisplay.className = "";
                        }
                    }, 1000);

                } else {
                    gameState.lastPitchResult = 'FOUL';
                    gameState.consecutivePerfectHits = 0;
                    gameState.consecutiveFouls++;

                    if (gameState.consecutiveFouls >= 4) {
                        soundManager.playBatBreak();
                        if (gameState.settings?.vibration && navigator.vibrate) navigator.vibrate([100, 50, 100, 50, 300]);

                        document.body.style.animation = "shake 0.5s cubic-bezier(.36,.07,.19,.97) both";
                        setTimeout(() => { document.body.style.animation = ""; }, 500);

                        feedbackDisplay.innerText = "BAT BROKEN!";
                        feedbackDisplay.className = "foul-text";

                        setTimeout(() => {
                            gameOver();
                            startScreen.innerHTML = `
                                <span style="color: #ff0000; font-size: 1.0em;">BAT BROKEN!</span><br>
                                <span style="font-size: 1.0em; display: block; margin-top: 10px; animation: none; color: #fff;">SCORE: ${gameState.score}</span>
                                <span id="restart-msg" style="font-size: 0.8em; display: block; margin-top: 20px; animation: ${gameState.settings?.noFlash ? 'none' : 'blink 1s infinite'}; color: #fff;">TAP TO RETRY</span>
                            `;
                        }, 500);

                        ball.speedX = 0;
                        ball.speedY = 5;
                        ball.hit = true;
                        return;
                    }

                    let foulAngle;
                    const rand = Math.random();

                    if (rand < 0.3) {
                        const offset = 0.08 + Math.random() * 0.05;
                        if (Math.random() > 0.5) {
                            foulAngle = FAIR_ANGLE_RIGHT + offset;
                            ball.spin = 0.03 + (Math.random() * 0.03);
                        } else {
                            foulAngle = FAIR_ANGLE_LEFT - offset;
                            ball.spin = -(0.03 + (Math.random() * 0.03));
                        }

                    } else if (rand < 0.7) {
                        const offset = 0.25 + Math.random() * 0.5;
                        if (Math.random() > 0.5) {
                            foulAngle = FAIR_ANGLE_RIGHT + offset;
                            ball.spin = 0.05 + (Math.random() * 0.05);
                        } else {
                            foulAngle = FAIR_ANGLE_LEFT - offset;
                            ball.spin = -(0.05 + (Math.random() * 0.05));
                        }

                    } else {
                        if (Math.random() > 0.5) {
                            foulAngle = 0.2 + (Math.random() * 1.0);
                            ball.spin = 0.02;
                        } else {
                            foulAngle = Math.PI - (0.2 + (Math.random() * 1.0));
                            ball.spin = -0.02;
                        }
                    }

                    ball.speedX = Math.cos(foulAngle) * HIT_SPEED;
                    ball.speedY = Math.sin(foulAngle) * HIT_SPEED;

                    soundManager.playHit();

                    setTimeout(() => {
                        soundManager.playFoul();

                        let foulText = "FOUL";
                        if (hitAngle > -1.3) foulText = "FOUL<br>(LATE SWING)";
                        else if (hitAngle < -1.8) foulText = "FOUL<br>(EARLY SWING)";

                        feedbackDisplay.innerHTML = foulText;
                        feedbackDisplay.className = "foul-text";

                        if (gameState.strikes < 2) {
                            gameState.strikes++;
                        }
                    }, 600);

                    setTimeout(() => {
                        if (feedbackDisplay.innerText.includes("FOUL")) {
                            feedbackDisplay.innerText = "";
                            feedbackDisplay.className = "";
                        }
                    }, 2000);
                }

                ball.hit = true;
                break;
            }
        }
    }
}
