import { gameState } from './gameState.js';

// Sound Manager
class SoundManager {
    constructor() {
        this.audioContext = null;
        this.initialized = false;
        this.themeOscillators = [];
        this.themeTimer = null;
        this.themePlaying = false;
        this.noiseBuffer = null;
        this.masterGain = null;
    }

    init() {
        if (this.initialized) return Promise.resolve();

        // Add listeners for iOS auto-resume
        // Add listeners for iOS auto-resume
        document.addEventListener('visibilitychange', () => this._handleVisibilityChange());
        window.addEventListener('pageshow', () => this._handleVisibilityChange());
        document.addEventListener('touchstart', () => this._handleInteraction());
        document.addEventListener('click', () => this._handleInteraction());

        return new Promise((resolve) => {
            try {
                this.audioContext = new (window.AudioContext || window.webkitAudioContext)();
                this.masterGain = this.audioContext.createGain();
                this.masterGain.connect(this.audioContext.destination);

                // Resume immediately
                this.audioContext.resume().then(() => {
                    // Double check state, sometimes it needs a moment
                    if (this.audioContext.state === 'running') {
                        this._finishInit();
                        resolve();
                    } else {
                        // Poll for running state if not immediate
                        let attempts = 0;
                        const checkState = setInterval(() => {
                            attempts++;
                            if (this.audioContext.state === 'running' || attempts > 50) { // Stop after ~500ms
                                clearInterval(checkState);
                                this._finishInit();
                                resolve();
                            }
                        }, 10);
                    }
                });

            } catch (e) {
                console.warn('Web Audio API not supported');
                resolve(); // Resolve anyway to let game start
            }
        });
    }

    _handleVisibilityChange() {
        if (document.hidden) return;

        this._checkAndResume();
    }

    _handleInteraction() {
        this._checkAndResume();
    }

    _checkAndResume() {
        if (this.audioContext && (this.audioContext.state === 'suspended' || this.audioContext.state === 'interrupted')) {
            this.audioContext.resume().catch(() => { });
        }
    }

    _finishInit() {
        if (this.initialized) return;

        // Warm up with a silent buffer
        const buffer = this.audioContext.createBuffer(1, 1, 22050);
        const source = this.audioContext.createBufferSource();
        source.buffer = buffer;
        source.connect(this.audioContext.destination);
        source.start(0);

        // Keep-alive oscillator: Plays silent sound to keep audio thread active
        const keepAliveOsc = this.audioContext.createOscillator();
        const keepAliveGain = this.audioContext.createGain();
        keepAliveOsc.type = 'sine';
        keepAliveOsc.frequency.value = 1; // 1Hz (inaudible)
        keepAliveGain.gain.value = 0.0001; // Nearly silent
        keepAliveOsc.connect(keepAliveGain);
        keepAliveGain.connect(this.audioContext.destination);
        keepAliveOsc.start(0);

        this.initialized = true;
    }

    playSwing() {
        if (gameState.settings?.muteSound) return;
        if (!this.initialized) return;
        const ctx = this.audioContext;
        const now = ctx.currentTime;

        const osc = ctx.createOscillator();
        const gain = ctx.createGain();

        osc.type = 'sawtooth';
        osc.frequency.setValueAtTime(200, now);
        osc.frequency.exponentialRampToValueAtTime(100, now + 0.1);

        gain.gain.setValueAtTime(0.15, now);
        gain.gain.exponentialRampToValueAtTime(0.01, now + 0.1);

        osc.connect(gain);
        gain.connect(this.masterGain);

        osc.start(now);
        osc.stop(now + 0.1);
    }

    playPause(isPausing) {
        if (gameState.settings?.muteSound) return;
        if (!this.initialized) return;
        const ctx = this.audioContext;
        const now = ctx.currentTime;

        const osc = ctx.createOscillator();
        const gain = ctx.createGain();

        osc.type = 'square';

        if (isPausing) {
            // Stepped Descending tone for Pause (High -> Low)
            osc.frequency.setValueAtTime(660, now);
            osc.frequency.setValueAtTime(440, now + 0.1);
        } else {
            // Stepped Ascending tone for Unpause (Low -> High)
            osc.frequency.setValueAtTime(440, now);
            osc.frequency.setValueAtTime(660, now + 0.1);
        }

        gain.gain.setValueAtTime(0.1, now);
        gain.gain.linearRampToValueAtTime(0.01, now + 0.2);

        osc.connect(gain);
        gain.connect(this.masterGain);

        osc.start(now);
        osc.stop(now + 0.2);
    }

    playPerfectHit() {
        if (gameState.settings?.muteSound) return;
        if (!this.initialized) return;
        const ctx = this.audioContext;
        const now = ctx.currentTime;

        // Major Triad Arpeggio (C Major: C5, E5, G5, C6)
        // 8-bit style: Square waves, short duration
        const notes = [523.25, 659.25, 783.99, 1046.50];
        const duration = 0.1;
        const spacing = 0.08;

        notes.forEach((freq, index) => {
            const osc = ctx.createOscillator();
            const gain = ctx.createGain();
            const startTime = now + (index * spacing);

            osc.type = 'square';
            osc.frequency.value = freq;

            gain.gain.setValueAtTime(0.2, startTime);
            gain.gain.exponentialRampToValueAtTime(0.01, startTime + duration);

            osc.connect(gain);
            gain.connect(this.masterGain);

            osc.start(startTime);
            osc.stop(startTime + duration);
        });

        // Add a sparkly high finish
        const sparkle = ctx.createOscillator();
        const sparkleGain = ctx.createGain();
        const sparkleStart = now + (notes.length * spacing);

        sparkle.type = 'triangle';
        sparkle.frequency.setValueAtTime(1200, sparkleStart);
        sparkle.frequency.linearRampToValueAtTime(2000, sparkleStart + 0.3);

        sparkleGain.gain.setValueAtTime(0.1, sparkleStart);
        sparkleGain.gain.linearRampToValueAtTime(0.01, sparkleStart + 0.3);

        sparkle.connect(sparkleGain);
        sparkleGain.connect(this.masterGain);

        sparkle.start(sparkleStart);
        sparkle.stop(sparkleStart + 0.3);
    }

        // OPTIMIZED: "Celestial Arpeggio" (Monophonic to prevent stutter)
        // Instead of 5 parallel oscillators, we use 1 jumping oscillator.
        const osc = ctx.createOscillator();
        const gain = ctx.createGain();

        // F Major 7 Arpeggio: F4, A4, C5, E5
        const notes = [349.23, 440.00, 523.25, 659.25];
        const noteDur = 0.08;

        osc.type = 'triangle'; // Warm tone

        // Schedule frequencies on ONE oscillator
        notes.forEach((freq, i) => {
            osc.frequency.setValueAtTime(freq, now + (i * noteDur));
        });

        // Envelope: Sustain then Fade
        gain.gain.setValueAtTime(0.1, now);
        gain.gain.setValueAtTime(0.1, now + (notes.length * noteDur));
        gain.gain.exponentialRampToValueAtTime(0.01, now + (notes.length * noteDur) + 0.3);

        osc.connect(gain);
        gain.connect(this.masterGain);

        osc.start(now);
        osc.stop(now + (notes.length * noteDur) + 0.3);

        // Lightweight Sweep (Optional, simplified)
        // Removed heavy sweep to prioritize performance.
    }

    playZigZagCharge() {
        if (gameState.settings?.muteSound) return;
        if (!this.initialized) return;
        const ctx = this.audioContext;
        const now = ctx.currentTime;

        // "Alien Warble" - Sine wave with Frequency Modulation (Zig-Zagging pitch)
        const osc = ctx.createOscillator();
        const mainGain = ctx.createGain();

        const modulator = ctx.createOscillator();
        const modGain = ctx.createGain();

        // Carrier: Pure Sine (Softeset tone)
        osc.type = 'sine';
        osc.frequency.setValueAtTime(400, now);
        osc.frequency.linearRampToValueAtTime(600, now + 0.8); // Slight rise

        // Modulator: Square wave ("Zig Zag" shape)
        modulator.type = 'square';
        modulator.frequency.value = 12; // 12Hz speed (Fast wobble)

        // Modulation Depth: How far frequency jumps
        modGain.gain.value = 50; // Jumps +/- 50Hz

        // FM Synthesis: LFO -> ModGain -> Carrier Frequency
        modulator.connect(modGain);
        modGain.connect(osc.frequency);

        // Amplitude Envelope (Volume)
        mainGain.gain.setValueAtTime(0.15, now); // Increased to match standard (Sine needs more gain than Square)
        mainGain.gain.linearRampToValueAtTime(0.01, now + 0.8);

        osc.connect(mainGain);
        mainGain.connect(this.masterGain);

        osc.start(now);
        modulator.start(now);
        osc.stop(now + 0.8);
        modulator.stop(now + 0.8);
    }

    playTheme() {
        if (gameState.settings?.muteMusic) return;
        if (!this.initialized || this.themePlaying) return;
        this.themePlaying = true;
        const ctx = this.audioContext;

        // "Take Me Out to the Ball Game" - Simple 8-bit melody
        // Notes: C4, C5, A4, G4, E4, G4, D4
        const melody = [
            { note: 261.63, dur: 0.4 }, // Take
            { note: 523.25, dur: 0.4 }, // Me
            { note: 440.00, dur: 0.4 }, // Out
            { note: 392.00, dur: 0.4 }, // To
            { note: 329.63, dur: 0.4 }, // The
            { note: 392.00, dur: 0.4 }, // Ball
            { note: 293.66, dur: 1.2 }, // Game
            { note: 0, dur: 0.4 }       // Rest
        ];

        let noteIndex = 0;
        let nextNoteTime = ctx.currentTime;

        const playNextNote = () => {
            if (!this.themePlaying) return;

            const { note, dur } = melody[noteIndex];

            if (note > 0) {
                const osc = ctx.createOscillator();
                const gain = ctx.createGain();

                osc.type = 'square'; // 8-bit sound
                osc.frequency.setValueAtTime(note, nextNoteTime);

                gain.gain.setValueAtTime(0.05, nextNoteTime);
                gain.gain.exponentialRampToValueAtTime(0.01, nextNoteTime + dur - 0.05);

                osc.connect(gain);
                gain.connect(this.masterGain);

                osc.start(nextNoteTime);
                osc.stop(nextNoteTime + dur);
            }

            nextNoteTime += dur;
            noteIndex = (noteIndex + 1) % melody.length;

            // Schedule next check slightly before the note ends to keep timing tight
            // but we use AudioContext time for the actual audio scheduling so it stays in sync
            const delay = (nextNoteTime - ctx.currentTime) * 1000;
            this.themeTimer = setTimeout(playNextNote, Math.max(0, delay - 20));
        };

        playNextNote();
    }

    stopTheme() {
        this.themePlaying = false;
        if (this.themeTimer) {
            clearTimeout(this.themeTimer);
            this.themeTimer = null;
        }
    }

            gain.gain.setValueAtTime(0.2, now + startTime);
            gain.gain.exponentialRampToValueAtTime(0.01, now + startTime + 0.15);

            gain.connect(this.masterGain);

            osc.stop(now + startTime + 0.15);
        });
    }

    playThud() {
        if (gameState.settings?.muteSound) return;
        if (!this.initialized) return;

        const ctx = this.audioContext;
        const now = ctx.currentTime;

        // "SEISMIC GROWL" (Unique Scary Step)
        // Instead of a thump/knock, we use a dissonant cluster of low saws
        // creating a mechanical/beast-like vibration.

        // Frequencies: Low G (49Hz), Ab (52Hz - dissonance), and Octave (98Hz)
        const freqs = [45, 48, 90];

        freqs.forEach(f => {
            const osc = ctx.createOscillator();
            const gain = ctx.createGain();
            const filter = ctx.createBiquadFilter();

            osc.type = 'sawtooth'; // Gritty texture
            osc.frequency.value = f;

            // Heavy Lowpass to remove the "Buzz" and keep the "Growl"
            filter.type = 'lowpass';
            filter.frequency.value = 180;

            // Swell envelope (Slower attack = Mass)
            gain.gain.setValueAtTime(0, now);
            gain.gain.linearRampToValueAtTime(0.3, now + 0.05); // Little swell
            gain.gain.exponentialRampToValueAtTime(0.01, now + 0.25); // Fade out

            osc.connect(filter);
            filter.connect(gain);
            gain.connect(this.masterGain);

            osc.start(now);
            osc.stop(now + 0.3);
        });

        // Add a Sub-Sine for pure weight
        const sub = ctx.createOscillator();
        const subGain = ctx.createGain();
        sub.type = 'sine';
        sub.frequency.value = 30; // Deep rumble
        subGain.gain.setValueAtTime(0.5, now);
        subGain.gain.exponentialRampToValueAtTime(0.01, now + 0.3);

        sub.connect(subGain);
        subGain.connect(this.masterGain);
        sub.start(now);
        sub.stop(now + 0.3);
    }

    playRoar() {
        if (gameState.settings?.muteSound) return;
        if (!this.initialized) return;
        const ctx = this.audioContext;
        const now = ctx.currentTime;

        // "METALLIC BEAST ROAR"
        // Uses FM Synthesis: Modulator -> Carrier
        // Modulator is High Freq/High Gain to create "Metal" texture

        const carrier = ctx.createOscillator();
        const modulator = ctx.createOscillator();
        const modGain = ctx.createGain();
        const mainGain = ctx.createGain();

        // Carrier: Low Pitch Drop
        carrier.type = 'sawtooth';
        carrier.frequency.setValueAtTime(150, now);
        carrier.frequency.exponentialRampToValueAtTime(50, now + 1.5); // Long growl

        // Modulator: Creates the "Growl/Metal" texture
        modulator.type = 'square';
        modulator.frequency.value = 45; // Roughly sync w/ footsteps

        // FM Depth
        modGain.gain.setValueAtTime(200, now); // Intense modulation
        modGain.gain.exponentialRampToValueAtTime(50, now + 1.5);

        modulator.connect(modGain);
        modGain.connect(carrier.frequency);

        // Volume Envelope
        mainGain.gain.setValueAtTime(0, now);
        mainGain.gain.linearRampToValueAtTime(0.6, now + 0.1);
        mainGain.gain.exponentialRampToValueAtTime(0.01, now + 1.5);

        carrier.connect(mainGain);
        mainGain.connect(this.masterGain);

        carrier.start(now);
        modulator.start(now);
        carrier.stop(now + 1.5);
        modulator.stop(now + 1.5);
    }
    playBatBreak() {
        if (gameState.settings?.muteSound) return;
        if (!this.initialized) return;
        const ctx = this.audioContext;
        const now = ctx.currentTime;

        // "Wood Snap" - Sawtooth burst + Noise
        const osc = ctx.createOscillator();
        const gain = ctx.createGain();

        osc.type = 'sawtooth';
        osc.frequency.setValueAtTime(150, now);
        osc.frequency.exponentialRampToValueAtTime(50, now + 0.1);

        gain.gain.setValueAtTime(0.5, now);
        gain.gain.exponentialRampToValueAtTime(0.01, now + 0.1);

        osc.connect(gain);
        gain.connect(this.masterGain);

        osc.start(now);
        osc.stop(now + 0.1);

        // Noise Burst (Simulated with random modulation)
        const noiseOsc = ctx.createOscillator();
        const noiseGain = ctx.createGain();
        noiseOsc.type = 'square';
        noiseOsc.frequency.value = 1000; // High freq "crack"
        noiseGain.gain.setValueAtTime(0.3, now);
        noiseGain.gain.exponentialRampToValueAtTime(0.01, now + 0.2);

        noiseOsc.connect(noiseGain);
        noiseGain.connect(this.masterGain);
        noiseOsc.start(now);
        noiseOsc.stop(now + 0.2);
    }

    _createNoiseBuffer() {
        if (this.noiseBuffer) return;
        const bufferSize = this.audioContext.sampleRate * 2; // 2 seconds of noise
        const buffer = this.audioContext.createBuffer(1, bufferSize, this.audioContext.sampleRate);
        const data = buffer.getChannelData(0);
        for (let i = 0; i < bufferSize; i++) {
            data[i] = Math.random() * 2 - 1;
        }
        this.noiseBuffer = buffer;
    }


    playFirework() {
        if (gameState.settings?.muteSound) return;
        if (!this.initialized) return;
        if (!this.noiseBuffer) this._createNoiseBuffer();

        const ctx = this.audioContext;
        const now = ctx.currentTime;

        // 1. 8-Bit Square Wave Drop (Higher pitch "Winning" sound)
        const osc = ctx.createOscillator();
        const oscGain = ctx.createGain();

        osc.type = 'square';
        osc.frequency.setValueAtTime(600, now); // Start high!
        osc.frequency.exponentialRampToValueAtTime(150, now + 0.4); // Drop, but stay audible

        oscGain.gain.setValueAtTime(0.3, now);
        oscGain.gain.exponentialRampToValueAtTime(0.01, now + 0.3);

        osc.connect(oscGain);
        oscGain.connect(this.masterGain);

        osc.start(now);
        osc.stop(now + 0.4);

        // 2. Noise Burst (Brighter crunch)
        const noiseSource = ctx.createBufferSource();
        noiseSource.buffer = this.noiseBuffer;

        const noiseGain = ctx.createGain();
        noiseGain.gain.setValueAtTime(0.4, now);
        noiseGain.gain.exponentialRampToValueAtTime(0.01, now + 0.3);

        // Brighter filter
        const filter = ctx.createBiquadFilter();
        filter.type = 'lowpass';
        filter.frequency.value = 5000; // Let more highs through

        noiseSource.connect(filter);
        filter.connect(noiseGain);
        noiseGain.connect(this.masterGain);

        noiseSource.start(now);
        noiseSource.stop(now + 0.3);
    }

    playSettingsOpen() {
        if (gameState.settings?.muteSound) return;
        if (!this.initialized) return;
        const ctx = this.audioContext;
        const now = ctx.currentTime;

        const osc = ctx.createOscillator();
        const gain = ctx.createGain();

        osc.type = 'square';
        // Descending blip (Mid -> Low) - Matches Pause "Power Down"
        osc.frequency.setValueAtTime(500, now);
        osc.frequency.linearRampToValueAtTime(300, now + 0.1);

        gain.gain.setValueAtTime(0.1, now);
        gain.gain.linearRampToValueAtTime(0.01, now + 0.1);

        osc.connect(gain);
        gain.connect(this.masterGain);

        osc.start(now);
        osc.stop(now + 0.1);
    }

        gain.gain.setValueAtTime(0.1, now);
        gain.gain.linearRampToValueAtTime(0.01, now + 0.1);

    playUnlockablesOpen() {
        if (gameState.settings?.muteSound) return;
        this.playSettingsOpen(); // Reuse open sound
    }

    playUnlockablesClose() {
        if (gameState.settings?.muteSound) return;
        this.playSettingsClose(); // Reuse close sound
    }

    _playBopSound(now, volume = 0.1) {
        const ctx = this.audioContext;
        const osc = ctx.createOscillator();
        const gain = ctx.createGain();

        // Add small lookahead to avoid main thread jank
        const startTime = now;

        osc.type = 'sine';
        // Gentle "bop"
        osc.frequency.setValueAtTime(400, startTime);
        osc.frequency.exponentialRampToValueAtTime(300, startTime + 0.3);

        gain.gain.setValueAtTime(volume, startTime);
        gain.gain.exponentialRampToValueAtTime(0.01, startTime + 0.3);

        osc.connect(gain);
        gain.connect(this.masterGain);

        osc.start(startTime);
        osc.stop(startTime + 0.35);
    }

    playItemAlreadyEquipped() {
        if (gameState.settings?.muteSound) return;
        if (!this.initialized) return;

        const ctx = this.audioContext;
        if (ctx.state === 'suspended') {
            ctx.resume();
        }
        this._playBopSound(ctx.currentTime, 0.1);
    }

    playEquip() {
        if (gameState.settings?.muteSound) return;

        // Try to play even if not fully "initialized" flag is set, as long as we have context
        if (!this.audioContext && !this.initialized) return;

        const ctx = this.audioContext;
        if (ctx && ctx.state === 'suspended') {
            ctx.resume();
        }

        if (!ctx) return;

        // Boost volume significantly to match perceived loudness
        this._playBopSound(ctx.currentTime, 0.3);
    }

    playZigZag() {
        if (gameState.settings?.muteSound) return;
        if (!this.initialized) return;
        const ctx = this.audioContext;
        const now = ctx.currentTime;

        // "Zap" effect - Aggressive Square Wave
        const osc = ctx.createOscillator();
        const gain = ctx.createGain();
        const mod = ctx.createOscillator();
        const modGain = ctx.createGain();

        osc.type = 'square'; // Grittier sound
        osc.frequency.setValueAtTime(800, now); // Start higher
        osc.frequency.exponentialRampToValueAtTime(100, now + 0.5); // Deep drop

        // Extreme modulation
        mod.type = 'sawtooth';
        mod.frequency.value = 60; // Faster buzz
        modGain.gain.value = 500; // Huge depth

        mod.connect(modGain);
        modGain.connect(osc.frequency);

        gain.gain.setValueAtTime(0.3, now); // Louder
        gain.gain.exponentialRampToValueAtTime(0.01, now + 0.5);

        osc.connect(gain);
        gain.connect(this.masterGain);

        osc.start(now);
        osc.stop(now + 0.5);
        mod.start(now);
        mod.stop(now + 0.5);
    }





    playPurchase() {
        if (gameState.settings?.muteSound) return;
        if (!this.initialized) return;
        const ctx = this.audioContext;
        const now = ctx.currentTime;

        // "Cha-Ching!" + Fanfare
        // 1. Coin Jingle (High pitched rapid arpeggio)
        const coinNotes = [1200, 1500, 1800, 2200];
        coinNotes.forEach((freq, i) => {
            const osc = ctx.createOscillator();
            const gain = ctx.createGain();
            const t = now + (i * 0.05);

            osc.type = 'sine'; // Pure tone
            osc.frequency.setValueAtTime(freq, t);

            gain.gain.setValueAtTime(0.1, t);
            gain.gain.exponentialRampToValueAtTime(0.01, t + 0.1);

            osc.connect(gain);
            gain.connect(this.masterGain);
            osc.start(t);
            osc.stop(t + 0.1);
        });

        // 2. Major Chord Swell (Rewarding)
        const chordNotes = [523.25, 659.25, 783.99, 1046.50]; // C Major
        chordNotes.forEach((freq, i) => {
            const osc = ctx.createOscillator();
            const gain = ctx.createGain();

            osc.type = 'triangle'; // Softer, fuller sound
            osc.frequency.setValueAtTime(freq, now);

            gain.gain.setValueAtTime(0.0, now);
            gain.gain.linearRampToValueAtTime(0.1, now + 0.1); // Swell in
            gain.gain.exponentialRampToValueAtTime(0.01, now + 0.6); // Fade out

            osc.connect(gain);
            gain.connect(this.masterGain);
            osc.start(now);
            osc.stop(now + 0.6);
        });

        // 3. Sparkle Finish
        const sparkle = ctx.createOscillator();
        const sparkleGain = ctx.createGain();
        sparkle.type = 'square';
        sparkle.frequency.setValueAtTime(2000, now + 0.2);
        sparkle.frequency.linearRampToValueAtTime(4000, now + 0.5);

        sparkleGain.gain.setValueAtTime(0.05, now + 0.2);
        sparkleGain.gain.linearRampToValueAtTime(0.0, now + 0.5);

        sparkle.connect(sparkleGain);
        sparkleGain.connect(this.masterGain);
        sparkle.start(now + 0.2);
        sparkle.stop(now + 0.5);
    }

    playError() {
        if (gameState.settings?.muteSound) return;
        if (!this.initialized) return;
        const ctx = this.audioContext;
        const now = ctx.currentTime;

        // Low "Buzz" error
        const osc = ctx.createOscillator();
        const gain = ctx.createGain();

        osc.type = 'sawtooth';
        osc.frequency.setValueAtTime(150, now);
        osc.frequency.linearRampToValueAtTime(100, now + 0.15);

        gain.gain.setValueAtTime(0.2, now);
        gain.gain.exponentialRampToValueAtTime(0.01, now + 0.15);

        osc.connect(gain);
        gain.connect(this.masterGain);

        osc.start(now);
        osc.stop(now + 0.15);
    }

    playThunder() {
        if (gameState.settings?.muteSound) return;
        if (!this.initialized) return;
        if (!this.noiseBuffer) this._createNoiseBuffer();

        const ctx = this.audioContext;
        const now = ctx.currentTime;

        const noise = ctx.createBufferSource();
        noise.buffer = this.noiseBuffer;

        const filter = ctx.createBiquadFilter();
        const gain = ctx.createGain();

        filter.type = 'lowpass';
        filter.frequency.setValueAtTime(800, now); // Start rumbly
        filter.frequency.exponentialRampToValueAtTime(100, now + 1.5); // Fade to deep rumble

        gain.gain.setValueAtTime(1.0, now);
        gain.gain.exponentialRampToValueAtTime(0.01, now + 1.5);

        noise.connect(filter);
        filter.connect(gain);
        gain.connect(this.masterGain);

        noise.start(now);
        noise.stop(now + 1.5);
    }
